Drag & Drop

Directives

  • draggable element to drag
  • dropable area(s) to drop the element into

Advanced Features

  • Placeholder where the element is placed if dropped

draggable

Link Function: 

angular.element(element).attr('draggable', 'true');

element.bind('dragstart', function () {
    var storyId = angular.element(element).attr('story-id');
    localStorage.setItem('draggedStoryId', storyId);
    $rootScope.$emit('storyDraggingStart');
});

element.bind('dragend', function () {
    $rootScope.$emit('storyDraggingEnd');
});

dropable

Link Function: 

var id = attrs.id; // generate id, if no id exists

element.bind('drop', function (e) {
    e.stopPropagation(); e.preventDefault();
    
    var src = getStory(storyId);
    var dest = document.getElementById(id);
    var storyId = localStorage.getItem('draggedStoryId');
    
    $rootScope.$broadcast('story-dragged', src, dest, storyId);
});
